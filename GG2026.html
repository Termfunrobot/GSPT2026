Arduino Firmware Code (PRO V4.0 เช็คผนังซ้าย/ขวา หมุนกลับด้านเมื่อโดนข้างเดียว) 

#include <EEPROM.h>
#include <Servo.h>

const int BUZZER_PIN = 13; 
const int START_BTN = A0;  

const int SERVO_PIN1 = 2; 
const int SERVO_PIN2 = 3; 

const int SENSOR_R1 = A1; // ขวาตัวหน้า
const int SENSOR_L1 = A2; // ซ้ายตัวหน้า
const int SENSOR_R2 = A3; // ขวาตัวหลัง
const int SENSOR_L2 = A4; // ซ้ายตัวหลัง

const int M1_IN1 = 8;  const int M1_IN2 = 9;
const int M2_IN1 = 10; const int M2_IN2 = 11;
const int M3_IN1 = 4;  const int M3_IN2 = 5;
const int M4_IN1 = 6;  const int M4_IN2 = 7;

int defaultSpeed = 100;
int motorLeftRatio = 100;  
int motorRightRatio = 100; 

struct Command {
  char action[4]; 
  float value;    
  int extra1; 
  int extra2; 
};

struct RobotSettings {
  uint8_t magic; 
  int speed;
  int ml;
  int mr;
};

const int MAX_CMDS = 50; 
Command cmdQueue[MAX_CMDS];
uint8_t cmdCount = 0;

unsigned long lastReceiveTime = 0;
const unsigned long FINISH_TRANSFER_TIMEOUT = 1000; 
bool isReceivingBatch = false; 

Servo servo1;
Servo servo2;

void saveSettingsToEEPROM() {
  RobotSettings settings = {0xAA, defaultSpeed, motorLeftRatio, motorRightRatio};
  EEPROM.put(800, settings);
}

void setup() {
  Serial.begin(115200); 
  Serial.setTimeout(10); 
  
  pinMode(BUZZER_PIN, OUTPUT);
  pinMode(START_BTN, INPUT); 
  pinMode(SENSOR_R1, INPUT);
  pinMode(SENSOR_L1, INPUT);
  pinMode(SENSOR_R2, INPUT); 
  pinMode(SENSOR_L2, INPUT); 

  for(int i = 4; i <= 11; i++) pinMode(i, OUTPUT);
  servo1.attach(SERVO_PIN1);
  servo2.attach(SERVO_PIN2);
  servo1.write(45); 
  servo2.write(45); 

  RobotSettings settings;
  EEPROM.get(800, settings);
  if (settings.magic == 0xAA) {
    defaultSpeed = constrain(settings.speed, 10, 100);
    motorLeftRatio = constrain(settings.ml, 50, 100);
    motorRightRatio = constrain(settings.mr, 50, 100);
  }

  EEPROM.get(0, cmdCount);
  if (cmdCount == 255 || cmdCount > MAX_CMDS) cmdCount = 0;
  for(int i = 0; i < cmdCount; i++) {
    EEPROM.get(1 + i * sizeof(Command), cmdQueue[i]);
  }

  playTone(523, 100); delay(120); 
  playTone(659, 100); delay(120); 
  playTone(784, 200); delay(200); 
  Serial.println("Arduino Ready. Speed: " + String(defaultSpeed) + "% | Stored Commands: " + String(cmdCount));
}

void loop() {
  unsigned long currentTime = millis();
  if (Serial.available() > 0) {
    String data = Serial.readStringUntil('\n');
    data.trim(); 
    if (data.length() > 0) {
      if (data == "?SENS") {
        Serial.print("SENS:");
        Serial.print(analogRead(A1)); Serial.print(",");
        Serial.print(analogRead(A2)); Serial.print(",");
        Serial.print(analogRead(A3)); Serial.print(",");
        Serial.print(analogRead(A4)); Serial.print(",");
        Serial.println(analogRead(A5));
        return;
      }
      if (!isReceivingBatch) {
        cmdCount = 0;
        EEPROM.update(0, 0); 
        isReceivingBatch = true;
        Serial.println("--- New Batch Start ---");
      }
      lastReceiveTime = currentTime; 

      String actionStr = "";
      float val = 0; int ex1 = 0; int ex2 = 0;
      
      if (data.startsWith("SP")) { defaultSpeed = constrain(data.substring(2).toInt(), 10, 100); saveSettingsToEEPROM(); return; }
      else if (data.startsWith("ML")) { motorLeftRatio = constrain(data.substring(2).toInt(), 50, 100); saveSettingsToEEPROM(); return; }
      else if (data.startsWith("MR")) { motorRightRatio = constrain(data.substring(2).toInt(), 50, 100); saveSettingsToEEPROM(); return; }
      else if (data.startsWith("SL")) { actionStr = "SL"; val = data.substring(2).toFloat(); }
      else if (data.startsWith("SR")) { actionStr = "SR"; val = data.substring(2).toFloat(); }
      else if (data.startsWith("CWL")) { 
        actionStr = "CWL"; 
        int c1 = data.indexOf(','); int c2 = data.indexOf(',', c1 + 1);
        if (c1 != -1 && c2 != -1) { val = data.substring(3, c1).toFloat(); ex1 = data.substring(c1 + 1, c2).toInt(); ex2 = data.substring(c2 + 1).toInt(); } 
        else { val = data.substring(3).toFloat(); ex1 = 350; ex2 = 0; }
      }
      else if (data.startsWith("CWR")) { 
        actionStr = "CWR"; 
        int c1 = data.indexOf(','); int c2 = data.indexOf(',', c1 + 1);
        if (c1 != -1 && c2 != -1) { val = data.substring(3, c1).toFloat(); ex1 = data.substring(c1 + 1, c2).toInt(); ex2 = data.substring(c2 + 1).toInt(); } 
        else { val = data.substring(3).toFloat(); ex1 = 350; ex2 = 0; }
      }
      else if (data.startsWith("CW")) { 
        actionStr = "CW"; 
        int c1 = data.indexOf(','); int c2 = data.indexOf(',', c1 + 1);
        if (c1 != -1 && c2 != -1) { val = data.substring(2, c1).toFloat(); ex1 = data.substring(c1 + 1, c2).toInt(); ex2 = data.substring(c2 + 1).toInt(); } 
        else { val = data.substring(2).toFloat(); ex1 = 350; ex2 = 0; }
      }
      else if (data.startsWith("F")) { actionStr = "F"; val = data.substring(1).toFloat(); }
      else if (data.startsWith("B")) { actionStr = "B"; val = data.substring(1).toFloat(); }
      else if (data.startsWith("L")) { actionStr = "L"; val = data.substring(1).toFloat(); }
      else if (data.startsWith("R")) { actionStr = "R"; val = data.substring(1).toFloat(); }
      else if (data.startsWith("D1:")) { actionStr = "D1"; ex1 = data.substring(3).toInt(); }
      else if (data.startsWith("D2:")) { actionStr = "D2"; ex1 = data.substring(3).toInt(); }

      if (actionStr != "" && cmdCount < MAX_CMDS) {
        actionStr.toCharArray(cmdQueue[cmdCount].action, 4);
        cmdQueue[cmdCount].value = val; cmdQueue[cmdCount].extra1 = ex1; cmdQueue[cmdCount].extra2 = ex2;
        EEPROM.put(1 + cmdCount * sizeof(Command), cmdQueue[cmdCount]); 
        cmdCount++; EEPROM.put(0, cmdCount); 
        playTone(2000, 30); 
      }
    }
  }

  if (isReceivingBatch && (currentTime - lastReceiveTime > FINISH_TRANSFER_TIMEOUT)) {
    playTone(1500, 800); delay(1000); playTone(1500, 800); delay(1000);
    isReceivingBatch = false; 
  }

  if (analogRead(START_BTN) > 800) {
    delay(50); 
    if (analogRead(START_BTN) > 800) {
      if (cmdCount > 0) {
        playTone(1047, 1000); delay(1000); 
        executeAllStoredCommands();
        playTone(784, 200); delay(200); 
      }
      while(analogRead(START_BTN) > 800); 
    }
  }
}

void executeAllStoredCommands() {
  for(int i = 0; i < cmdCount; i++) {
    String act = String(cmdQueue[i].action);
    float val = cmdQueue[i].value; int ex1 = cmdQueue[i].extra1; int ex2 = cmdQueue[i].extra2;
    
    float baseDelayTime;
    if (act == "L" || act == "R") {
      baseDelayTime = (val / 80.0) * 250.0;
    } else {
      baseDelayTime = val * 5.0; 
    }

    float speedRatio = 100.0 / (float)defaultSpeed;
    float frictionBoost = 1.0;
    
    if (act == "L" || act == "R") {
      if (defaultSpeed < 100) {
        float turnBoostFactor = 0.08; 
        frictionBoost = 1.0 + ((100.0 - defaultSpeed) / 100.0) * turnBoostFactor;
      }
    } 
    else {
      frictionBoost = (defaultSpeed < 80) ? 1.04 : 1.0; 
    }
    
    int compensatedDelay = (int)(baseDelayTime * speedRatio * frictionBoost);
    
    if (act == "F") { movePulsed(1, 1, 1, 1, compensatedDelay, defaultSpeed); }
    else if (act == "B") { movePulsed(-1, -1, -1, -1, compensatedDelay, defaultSpeed); }
    else if (act == "L") { movePulsed(-1, -1, 1, 1, compensatedDelay, defaultSpeed); } 
    else if (act == "R") { movePulsed(1, 1, -1, -1, compensatedDelay, defaultSpeed); } 
    else if (act == "SL") { movePulsed(1, -1, 1, -1, compensatedDelay, defaultSpeed); }
    else if (act == "SR") { movePulsed(-1, 1, -1, 1, compensatedDelay, defaultSpeed); }
    else if (act == "D1") { servo1.write(ex1); delay(600); servo1.write(45); delay(400); }
    else if (act == "D2") { servo2.write(ex1); delay(600); servo2.write(45); delay(400); }
    else if (act == "CW") { checkWallSequence(ex1, ex2); }
    else if (act == "CWL") { checkWallLeftSequence(ex1, ex2); }
    else if (act == "CWR") { checkWallRightSequence(ex1, ex2); }
    
    stopMotors(); delay(300); 
  }
}

void movePulsed(int m1, int m2, int m3, int m4, int totalDelay, int speedPercent) {
  speedPercent = constrain(speedPercent, 0, 100);
  if (speedPercent == 0) { stopMotors(); delay(totalDelay); return; }

  int effectiveSpeedL = (speedPercent * motorLeftRatio) / 100;
  int effectiveSpeedR = (speedPercent * motorRightRatio) / 100;

  int onTime = 20; 
  int offTimeL = map(effectiveSpeedL, 10, 100, 50, 0);
  int offTimeR = map(effectiveSpeedR, 10, 100, 50, 0);
  if (offTimeL < 0) offTimeL = 0; if (offTimeR < 0) offTimeR = 0;

  unsigned long startTime = millis();
  while (millis() - startTime < (unsigned long)totalDelay) {
    unsigned long cycleStart = millis();
    bool leftDone = false; bool rightDone = false;
    setMotors(m1, m2, m3, m4);
    
    while (!leftDone || !rightDone) {
      unsigned long elapsed = millis() - cycleStart;
      if (elapsed >= (unsigned long)(onTime + offTimeL)) leftDone = true;
      else if (elapsed >= (unsigned long)onTime) {
        digitalWrite(M1_IN1, LOW); digitalWrite(M1_IN2, LOW);
        digitalWrite(M2_IN1, LOW); digitalWrite(M2_IN2, LOW);
      }

      if (elapsed >= (unsigned long)(onTime + offTimeR)) rightDone = true;
      else if (elapsed >= (unsigned long)onTime) {
        digitalWrite(M3_IN1, LOW); digitalWrite(M3_IN2, LOW);
        digitalWrite(M4_IN1, LOW); digitalWrite(M4_IN2, LOW);
      }
      if (millis() - startTime >= (unsigned long)totalDelay) break;
    }
  }
  stopMotors();
}

// เช็คผนังด้านหน้า/หลัง
void checkWallSequence(int threshold, int logic) {
  while (true) {
    int valL1 = analogRead(A2); int valL2 = analogRead(A4);
    int valR1 = analogRead(A1); int valR2 = analogRead(A3);

    bool leftHit = (logic == 0) ? (valL1 < threshold || valL2 < threshold) : (valL1 > threshold || valL2 > threshold);
    bool rightHit = (logic == 0) ? (valR1 < threshold || valR2 < threshold) : (valR1 > threshold || valR2 > threshold);

    if (leftHit && rightHit) { stopMotors(); delay(300); break; }
    else if (leftHit) { setMotors(1, 1, -1, -1); delay(10); stopMotors(); delay(50); }
    else if (rightHit) { setMotors(-1, -1, 1, 1); delay(10); stopMotors(); delay(50); }
    else { setMotors(-1, -1, -1, -1); delay(10); stopMotors(); delay(50); }
  }
  setMotors(1, 1, 1, 1); delay(100); stopMotors();
}

// -------------------------------------------------------------
// เช็คกำแพงซ้าย (เซ็นเซอร์ A2 หน้าซ้าย และ A4 หลังซ้าย)
// *หมุนกลับด้านตรงกันข้ามเมื่อเซ็นเซอร์โดนข้างเดียว*
// -------------------------------------------------------------
void checkWallLeftSequence(int threshold, int logic) {
  while (true) {
    int valA2 = analogRead(SENSOR_L1); // หน้าซ้าย
    int valA4 = analogRead(SENSOR_L2); // หลังซ้าย

    bool hitFront = (logic == 0) ? (valA2 < threshold) : (valA2 > threshold);
    bool hitRear  = (logic == 0) ? (valA4 < threshold) : (valA4 > threshold);

    if (hitFront && hitRear) { 
      // ขนานผนังซ้ายแล้ว หยุดทันที
      stopMotors(); delay(300); break; 
    }
    else if (hitFront) { 
      // โดนแค่ตัวหน้าซ้าย -> หมุนกลับด้านเป็น หมุนซ้าย (L)
      setMotors(-1, -1, 1, 1); delay(10); stopMotors(); delay(50); 
    } 
    else if (hitRear) { 
      // โดนแค่ตัวหลังซ้าย -> หมุนกลับด้านเป็น หมุนขวา (R)
      setMotors(1, 1, -1, -1); delay(10); stopMotors(); delay(50); 
    } 
    else { 
      // ยังไม่โดนผนัง -> สไลด์ซ้าย (SL) เข้าหาผนัง
      setMotors(1, -1, 1, -1); delay(10); stopMotors(); delay(50); 
    }
  }
  // สไลด์ขวา (SR) ถอยออกเล็กน้อย ไม่ให้เบียดผนังเกินไป
  setMotors(-1, 1, -1, 1); delay(100); stopMotors();
}

// -------------------------------------------------------------
// เช็คกำแพงขวา (เซ็นเซอร์ A1 หน้าขวา และ A3 หลังขวา)
// *หมุนกลับด้านตรงกันข้ามเมื่อเซ็นเซอร์โดนข้างเดียว*
// -------------------------------------------------------------
void checkWallRightSequence(int threshold, int logic) {
  while (true) {
    int valA1 = analogRead(SENSOR_R1); // หน้าขวา
    int valA3 = analogRead(SENSOR_R2); // หลังขวา

    bool hitFront = (logic == 0) ? (valA1 < threshold) : (valA1 > threshold);
    bool hitRear  = (logic == 0) ? (valA3 < threshold) : (valA3 > threshold);

    if (hitFront && hitRear) { 
      // ขนานผนังขวาแล้ว หยุดทันที
      stopMotors(); delay(300); break; 
    }
    else if (hitFront) { 
      // โดนแค่ตัวหน้าขวา -> หมุนกลับด้านเป็น หมุนขวา (R)
      setMotors(1, 1, -1, -1); delay(10); stopMotors(); delay(50); 
    } 
    else if (hitRear) { 
      // โดนแค่ตัวหลังขวา -> หมุนกลับด้านเป็น หมุนซ้าย (L)
      setMotors(-1, -1, 1, 1); delay(10); stopMotors(); delay(50); 
    } 
    else { 
      // ยังไม่โดนผนัง -> สไลด์ขวา (SR) เข้าหาผนัง
      setMotors(-1, 1, -1, 1); delay(10); stopMotors(); delay(50); 
    }
  }
  // สไลด์ซ้าย (SL) ถอยออกเล็กน้อย ไม่ให้เบียดผนังเกินไป
  setMotors(1, -1, 1, -1); delay(100); stopMotors();
}

void setMotors(int m1, int m2, int m3, int m4) {
  digitalWrite(M1_IN1, m1 == 1 ? HIGH : LOW); digitalWrite(M1_IN2, m1 == -1 ? HIGH : LOW);
  digitalWrite(M2_IN1, m2 == 1 ? HIGH : LOW); digitalWrite(M2_IN2, m2 == -1 ? HIGH : LOW);
  digitalWrite(M3_IN1, m3 == 1 ? HIGH : LOW); digitalWrite(M3_IN2, m3 == -1 ? HIGH : LOW);
  digitalWrite(M4_IN1, m4 == 1 ? HIGH : LOW); digitalWrite(M4_IN2, m4 == -1 ? HIGH : LOW);
}
void stopMotors() { setMotors(0, 0, 0, 0); }
void playTone(int freq, int duration) { tone(BUZZER_PIN, freq, duration); }
